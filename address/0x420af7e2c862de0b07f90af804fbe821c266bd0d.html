    constructor(address payable mktAddr, address payable prjAddr) {
        require(!isContract(mktAddr) && !isContract(prjAddr));
        mktAds = mktAddr;
        prjAds = prjAddr;
        CP = getContractBalanceRate();
        CPV = getContractBalanceRateVip();

function FeePayout(uint amt) internal{
    uint mktFee = amt.mul(MKTF).div(PD);
    uint prjFee = amt.mul(PJTF).div(PD);
    mktAds.transfer(mktFee);
    prjAds.transfer(prjFee);
    emit FeePayed(msg.sender, mktFee.add(prjFee));
}

function RefsPayout(uint msgValue) internal{
    User storage user = users[msg.sender];
    if (user.refr != address(0)) {
           address up = user.refr;
        for (uint i = 0; i < 3; i++) {
                if (up != address(0)) {
                    uint amt = msgValue.mul(DPTS[i]).div(PD);
                    if (amt > 0) {
                        users[up].db = uint64(uint(users[up].db).add(amt));
                    }
                    users[up].refs[i]++;
                    up = users[up].refr;
                } else break;
            }
        }
}

function PoolTopPts(uint msgValue) internal{
    User storage user = users[msg.sender];
    if (user.refr != address(0)) {
        address up = user.refr;
        purds[pC][up] = purds[pC][up].add(msgValue);
        for(uint i = 0; i < 4; i++) {
                if(ptop[i] == up) break;
                if(ptop[i] == address(0)) {
                ptop[i] = up;
                break;
        }    
            if(purds[pC][up] > purds[pC][ptop[i]]) {
                for(uint j = i + 1; j < 4; j++) {
                    if(ptop[j] == up) {
                        for(uint k = j; k <= 4; k++) {
                            ptop[k] = ptop[k + 1];
                        }
                        break;
                    }
                }
                for(uint j = uint(4 - 1); j > i; j--) {
                    ptop[j] = ptop[j - 1];
                }
                ptop[i] = up;
                break;
            }
        }
    }
}

    function invest(address refr) public payable {
        require(!isContract(msg.sender) && msg.sender == tx.origin);
        User storage user = users[msg.sender];
        require(msg.value >= IMA && user.deps.length < DPSM,"Out limits deposit");
        FeePayout (msg.value);
        if (user.refr == address(0) && users[refr].deps.length > 0 && refr != msg.sender) {
            user.refr = refr;
        }
	    RefsPayout(msg.value);
        resetpools();
        pB = pB.add(msg.value.mul(3).div(100));
        PoolTopPts(msg.value);
        if (user.deps.length == 0) {
            user.ckpt = uint32(block.timestamp);
            emit Newbie(msg.sender);
        }
        user.deps.push(Deps(uint64(msg.value), 0, uint32(block.timestamp)));
        tInv = tInv.add(msg.value);
        tDps++;
        if (CP < BP.add(MCP)) {
            uint contractPercentNew = getContractBalanceRate();
            if (contractPercentNew > CP) {
                CP = contractPercentNew;
            }
        }
        emit NewDeposit(msg.sender, msg.value);
    }

    function withdraw() public {
        uint cB = address(this).balance;
        User storage user = users[msg.sender];
        require (block.timestamp >= uint(user.ckpt).add(TS.mul(2)) && cB > 0, "Avaliable in 36hours");
        uint uPR = getUserPercentRate(msg.sender);
        uint tAmt;
        uint divs;

         for (uint i = 0; i < user.deps.length; i++) {
            if (uint(user.deps[i].wdn) < uint(user.deps[i].amt).mul(3).div(2)) {
                if (user.deps[i].st > user.ckpt) {
                    divs = (uint(user.deps[i].amt).mul(uPR).div(PD))
                        .mul(block.timestamp.sub(uint(user.deps[i].st)))
                        .div(TS);
                } else {
                    divs = (uint(user.deps[i].amt).mul(uPR).div(PD))
                        .mul(block.timestamp.sub(uint(user.ckpt)))
                        .div(TS);
                }
                if (uint(user.deps[i].wdn).add(divs) > uint(user.deps[i].amt).mul(3).div(2)) {
                    divs = (uint(user.deps[i].amt).mul(3).div(2)).sub(uint(user.deps[i].wdn));
                }
        
        if (uint(user.deps[i].wdn).add(divs) < uint(user.deps[i].amt).mul(3).div(2) && user.db > 0) {
            uint direct_bonus = user.db;
            if(uint(user.deps[i].wdn).add(divs).add(direct_bonus) > uint(user.deps[i].amt).mul(3).div(2)) {
                direct_bonus = (uint(user.deps[i].amt).mul(3).div(2)).sub(uint(user.deps[i].wdn).add(divs));
            }
            user.db = uint64(uint(user.db).sub(direct_bonus));
            divs = divs.add(direct_bonus);
            }
            
        if (uint(user.deps[i].wdn).add(divs) < uint(user.deps[i].amt).mul(3).div(2) && user.pb > 0) {
            uint pool_bonus = user.pb;
            if(uint(user.deps[i].wdn).add(divs).add(pool_bonus) > uint(user.deps[i].amt).mul(3).div(2)) {
                pool_bonus = (uint(user.deps[i].amt).mul(3).div(2)).sub(uint(user.deps[i].wdn).add(divs));
            }
            user.pb = uint64(uint(user.pb).sub(pool_bonus));
            divs = divs.add(pool_bonus);
            }
        
        user.deps[i].wdn = uint64(uint(user.deps[i].wdn).add(divs)); /// changing of storage data
        tAmt = tAmt.add(divs);
            }
        }
        require(tAmt > WITHDRAW_MIN_AMOUNT, "User no minimum");
            if (cB < tAmt) {
            tAmt = cB;
            }   
        user.ckpt = uint32(block.timestamp);
        msg.sender.transfer(tAmt);
        tWdn = tWdn.add(tAmt);
        emit Withdrawn(msg.sender, tAmt);
    }

        require(!isContract(msg.sender) && msg.sender == tx.origin);
        uint MinLimit = getticketcost();
        require(msg.value == MinLimit, "Check value ticket");

    if (aET < block.timestamp) {
            resetpools();
    }
    if (aH < msg.value){
            aH = msg.value;
        }
    if (aET.sub(5 minutes) < block.timestamp) {
            aET = (uint(aET).add(5 minutes));
    }
    aP = aP.add(msg.value);
    if (msg.sender != address(0)) {
           address up = msg.sender;
           auds[aC][up] = msg.value;
        for(uint i = 0; i < 7; i++) {
                if(at[i] == address(0)) {
                at[i] = up;
                break;
            }
            if(auds[aC][up] > auds[aC][at[i]]) {
                for(uint j = uint(7 - 1); j > i; j--) {
                    at[j] = at[j - 1];
                }
               at[i] = up;
                break;
            }
        }
        aPS++;
    }
    user.aparticipation++;
}

    function resetpools() public {
    if((uint(pLD)).add(TS) < block.timestamp) {
        uint da = pB.div(10);
       for(uint i = 0; i < 4; i++) {
            if(ptop[i] != address(0)){
            uint win = da.mul(PPTS[i]).div(100);
            users[ptop[i]].pb = uint64(uint(users[ptop[i]].pb).add(win));
            pB = pB.sub(win);
            emit PoolPayout(ptop[i], win);
            }
        }
        for(uint i = 0; i < 4; i++) {
            ptop[i] = address(0);
        }
        pLD = uint32(block.timestamp);
        pC++;
    }
    if (aET < block.timestamp) {
        if (aPS > 0){
            uint da = aP;
            if (aB > 0){
            da = da.add(aB.div(20));   
            }
            
            aB = aB.add(aP.div(10));
            
            FeePayout (aP);
            aLP = da;
            for(uint i = 0; i < 5; i++) {
            if(at[i] != address(0)){
            uint win = aLP.mul(APTCS[i]).div(100);
            address(uint160(at[i])).transfer(win);
            users[at[i]].apft = uint64(uint(users[at[i]].apft).add(win));
            da = da.sub(win);
            emit AuctionPayout(at[i], win);
            }
        }
        da = da.sub(aLP.mul(41).div(100));
        
        if (da > 0) {
                uint residual = da;
                aB = aB.add(da);
                da = da.sub(residual);
            }
        for(uint i = 0; i < 7; i++) {
            alt[i] = at[i];
            if(at[i] != address(0)){
            at[i] = address(0);
            }
        }
        aP = 0;
        aH = 0;
        aPS = 0;  
        }
        aST = uint32(block.timestamp);
        aET = (uint(aST).add(AS));
        aC++;
    }
    }
